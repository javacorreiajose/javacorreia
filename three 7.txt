                <ul>
                    <li>

                        <ul>
                            <li><input type="text" placeholder="Enter text here"></li>
                            <li><input type="text" placeholder="Enter text here"></li>
                        </ul>
                    </li>
                    <li><input type="text" placeholder="Enter text here"></li>
                </ul>
            </li>
            <li><input type="text" placeholder="Enter text here"></li>
        </ul>

    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const tree = document.querySelector('.tree');
            let textBoxCount = 7;  // Initialize with the number of existing text boxes

            function addTextBox(event) {
                if (textBoxCount < 7) {
                    const ul = document.createElement('ul');
                    const li = document.createElement('li');
                    const input = document.createElement('input');
                    input.type = 'text';
                    input.placeholder = 'Enter text here';

                    li.appendChild(input);
                    ul.appendChild(li);
                    event.target.parentNode.appendChild(ul);
                    textBoxCount++;
                } else {
                    alert('Limit of 7 text boxes reached.');
                }
            }

            tree.addEventListener('dblclick', (event) => {
                if (event.target.tagName === 'INPUT') {
                    addTextBox(event);
                }
            });
        });
    </script>
